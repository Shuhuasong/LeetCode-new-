
public int longestSubstring(String s){
   if(s==null || s.length()==0) return -1;
   int result = 0, n = s.length();
   int i=0;
   Set<Character> set = new HashSet<>();
   for(int j=0; j<n; ){
       char ch = s.charAt(j);
       if(set.add(ch)){
          j++;
          result = Math.max(result, j-i);
       }
       set.remove(s.charAt(i));
       i++;
   }
   return result;
}
